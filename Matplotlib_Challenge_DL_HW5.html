df_2 = pd.read_csv(clinical_trial_data_to_load)

df_main = pd.merge(df_1,df_2, on="Mouse ID", how ='right')

# Plot 1 : Tumor Response to Treatement
df_main_grpby = df_main.groupby(["Timepoint","Drug"]).mean()

# Plot 4 : Summary

percent_change = (df_main_1.loc[45] - df_main_1.loc[0])/df_main_1.loc[45]*100
percent_change = percent_change.to_frame()

percent_change.columns =["%Change"]

colors = []
for value in percent_change["%Change"]:
    if value > 0:
        colors.append('r')
    else:
        colors.append('g')

x_ticks = percent_change.index
y_value = percent_change["%Change"]


rects = plt.bar(x_ticks,y_value,width = .95, color=colors, alpha = 0.75, align="center")
plt.title("Tumor Change Over 45 Day Treatment")
plt.ylabel("% Tumor Volume Change")
plt.grid(alpha = 0.5)
plt.xticks(x_ticks,x_ticks, rotation=45)


for rect in rects:
    height = rect.get_height()
    plt.text(rect.get_x(), 3, str(round(height,0))+"%" ,size = 8, color ="white")
    plt.text(rect.get_x(), -3, str(round(height,0))+"%" ,size = 8, color ="white")
            
plt.show()
